CREATE TABLE IF NOT EXISTS libraries ( id INTEGER DEFAULT NULL PRIMARY KEY AUTOINCREMENT,name TEXT NOT NULL UNIQUE);
CREATE TABLE IF NOT EXISTS dependencies ( id_libraries INTEGER NOT NULL,reference INTEGER NOT NULL );
CREATE TABLE IF NOT EXISTS rules ( id_libraries INTEGER NOT NULL REFERENCES libraries (id),key TEXT NOT NULL ,value TEXT NOT NULL,required INTEGER DEFAULT 0 );
CREATE TABLE IF NOT EXISTS codes ( id_libraries INTEGER NOT NULL REFERENCES libraries (id),code BLOB NOT NULL,is_uri INTEGER DEFAULT 0 );
CREATE TABLE IF NOT EXISTS users ( id INTEGER DEFAULT NULL PRIMARY KEY AUTOINCREMENT,name TEXT NOT NULL UNIQUE,pwd TEXT NOT NULL,email TEXT NOT NULL,auth INTEGER NOT NULL REFERENCES users_auth (value) );
CREATE TABLE IF NOT EXISTS users_auth ( value INTEGER NOT NULL UNIQUE,description TEXT NOT NULL UNIQUE );
